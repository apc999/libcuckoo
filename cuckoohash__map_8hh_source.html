647  * hazard pointer, whithout locking anything. There is a
648  * possibility that after loading a snapshot and setting the
649  * hazard pointer, an expansion runs and create a new version of
650  * the table, leaving the old one for deletion. To deal with that,
651  * we check that the table_info we loaded is the same as the
652  * current one, and if it isn't, we try again. Whenever we check
653  * if (ti != table_info.load()) after setting the hazard pointer,

770  * hashed key could be. It takes the first possible bucket as a
771  * parameter. Note that this function will return the first
772  * possible bucket if index is the second possible bucket, so
773  * alt_index(ti, hv, alt_index(ti, hv, index_hash(ti, hv))) ==
774  * index_hash(ti, hv). */

804  /* b_slot holds the information for a BFS path through the

808  size_t bucket;

810  // buckets in the path.

813  // std::numeric_limits<decltype(pathcode)>::max(),

816  // occupies

823  /* b_queue is the queue used to store b_slots for BFS cuckoo
824  * hashing. */

826  b_slot slots[MAX_CUCKOO_COUNT+1];
827  size_t first;
828  size_t last;
829 
830  public:

834  void enqueue(b_slot x) {
835  slots[last] = x;
836  last = (last == MAX_CUCKOO_COUNT) ? 0 : last+1;
837  assert(last != first);
838  }

841  assert(first != last);

843  first = (first == MAX_CUCKOO_COUNT) ? 0 : first+1;
844  return x;
845  }

853  /* slot_search searches for a cuckoo path using breadth-first
854  search. It starts with the i1 and i2 buckets, and, until it finds
855  a bucket with an empty slot, adds each slot of the bucket in the
856  b_slot. If the queue runs out of space, it fails. */
857  static b_slot slot_search(const TableInfo *ti, const size_t i1,
858  const size_t i2) {
859  b_queue q;
860  // The initial pathcode informs cuckoopath_search which bucket
861  // the path starts on
862  q.enqueue(b_slot(i1, 0, 0));
863  q.enqueue(b_slot(i2, 1, 0));
864  while (q.not_full()) {
865  b_slot x = q.dequeue();
866  // Picks a random slot to start from
867  for (size_t slot = 0; slot < SLOT_PER_BUCKET && q.not_full(); slot++) {
868  lock(ti, x.bucket);
869  if (!ti->buckets_[x.bucket].occupied[slot]) {
870  // We can terminate the search here
871  x.pathcode = x.pathcode * SLOT_PER_BUCKET + slot;
872  unlock(ti, x.bucket);
873  return x;
874  }
875  // Create a new b_slot item, that represents the
876  // bucket we would look at after searching x.bucket
877  // for empty slots.
878  const size_t hv = hashed_key(ti->buckets_[x.bucket].keys[slot]);
879  unlock(ti, x.bucket);
880  b_slot y(alt_index(ti, hv, x.bucket),
881  x.pathcode * SLOT_PER_BUCKET + slot, x.depth+1);
882 
883  // Check if any of the slots in the prospective bucket
884  // are empty, and, if so, return that b_slot. We lock
885  // the bucket so that no changes occur while
886  // iterating.
887  lock(ti, y.bucket);
888  for (size_t j = 0; j < SLOT_PER_BUCKET; j++) {
889  if (!ti->buckets_[y.bucket].occupied.test(j)) {
890  y.pathcode = y.pathcode * SLOT_PER_BUCKET + j;
891  unlock(ti, y.bucket);
892  return y;
893  }
894  }
895  unlock(ti, y.bucket);
896 
897  // No empty slots were found, so we push this onto the
898  // queue
899  if (y.depth != MAX_BFS_DEPTH) {
900  q.enqueue(y);
901  }
902  }
903  }
904  // We didn't find a short-enough cuckoo path, so the queue ran
905  // out of space. Return a failure value.
906  return b_slot(0, 0, -1);
907  }

909  /* cuckoopath_search finds a cuckoo path from one of the starting
910  * buckets to an empty slot in another bucket. It returns the
911  * depth of the discovered cuckoo path on success, and -1 on
912  * failure. Since it doesn't take locks on the buckets it
913  * searches, the data can change between this function and
914  * cuckoopath_move. Thus cuckoopath_move checks that the data
915  * matches the cuckoo path before changing it. */
916  static int cuckoopath_search(const TableInfo *ti, CuckooRecord* cuckoo_path,
917  const size_t i1, const size_t i2) {
918  b_slot x = slot_search(ti, i1, i2);
919  if (x.depth == -1) {
920  return -1;
921  }
922  // Fill in the cuckoo path slots from the end to the beginning
923  for (int i = x.depth; i >= 0; i--) {
924  cuckoo_path[i].slot = x.pathcode % SLOT_PER_BUCKET;
925  x.pathcode /= SLOT_PER_BUCKET;
926  }
927  /* Fill in the cuckoo_path buckets and keys from the beginning
928  * to the end, using the final pathcode to figure out which
929  * bucket the path starts on. Since data could have been
930  * modified between slot_search and the computation of the
931  * cuckoo path, this could be an invalid cuckoo_path. */
932  CuckooRecord *curr = cuckoo_path;
933  if (x.pathcode == 0) {
934  curr->bucket = i1;
935  lock(ti, curr->bucket);
936  if (!ti->buckets_[curr->bucket].occupied[curr->slot]) {
937  // We can terminate here
938  unlock(ti, curr->bucket);
939  return 0;
940  }
941  curr->key = ti->buckets_[curr->bucket].keys[curr->slot];
942  unlock(ti, curr->bucket);
943  } else {
944  assert(x.pathcode == 1);
945  curr->bucket = i2;
946  lock(ti, curr->bucket);
947  if (!ti->buckets_[curr->bucket].occupied[curr->slot]) {
948  // We can terminate here
949  unlock(ti, curr->bucket);
950  return 0;
951  }
952  curr->key = ti->buckets_[curr->bucket].keys[curr->slot];
953  unlock(ti, curr->bucket);
954  }
955  for (int i = 1; i <= x.depth; i++) {
956  CuckooRecord *prev = curr++;
957  const size_t prevhv = hashed_key(prev->key);
958  assert(prev->bucket == index_hash(ti, prevhv) ||
959  prev->bucket == alt_index(ti, prevhv, index_hash(ti, prevhv)));
960  // We get the bucket that this slot is on by computing the
961  // alternate index of the previous bucket
962  curr->bucket = alt_index(ti, prevhv, prev->bucket);
963  lock(ti, curr->bucket);
964  if (!ti->buckets_[curr->bucket].occupied[curr->slot]) {
965  // We can terminate here
966  unlock(ti, curr->bucket);
967  return i;
968  }
969  curr->key = ti->buckets_[curr->bucket].keys[curr->slot];
970  unlock(ti, curr->bucket);
971  }
972  return x.depth;
973  }

976  /* cuckoopath_move moves keys along the given cuckoo path in order
977  * to make an empty slot in one of the buckets in cuckoo_insert.
978  * Before the start of this function, the two insert-locked
979  * buckets were unlocked in run_cuckoo. At the end of the
980  * function, if the function returns true (success), then the last
981  * bucket it looks at (which is either i1 or i2 in run_cuckoo)
982  * remains locked. If the function is unsuccessful, then both
983  * insert-locked buckets will be unlocked. */
984  static bool cuckoopath_move(TableInfo *ti, CuckooRecord* cuckoo_path,
985  size_t depth, const size_t i1, const size_t i2) {
986  if (depth == 0) {
987  /* There is a chance that depth == 0, when
988  * try_add_to_bucket sees i1 and i2 as full and
989  * cuckoopath_search finds one empty. In this case, we
990  * lock both buckets. If the bucket that cuckoopath_search
991  * found empty isn't empty anymore, we unlock them and
992  * return false. Otherwise, the bucket is empty and
993  * insertable, so we hold the locks and return true. */
994  const size_t bucket = cuckoo_path[0].bucket;
995  assert(bucket == i1 || bucket == i2);
996  lock_two(ti, i1, i2);
997  if (!ti->buckets_[bucket].occupied[cuckoo_path[0].slot]) {
998  return true;
999  } else {
1000  unlock_two(ti, i1, i2);
1001  return false;
1002  }
1003  }
1004 
1005  while (depth > 0) {
1006  CuckooRecord *from = cuckoo_path + depth - 1;
1007  CuckooRecord *to = cuckoo_path + depth;
1008  size_t fb = from->bucket;
1009  size_t fs = from->slot;
1010  size_t tb = to->bucket;
1011  size_t ts = to->slot;
1012 
1013  size_t ob = 0;
1014  if (depth == 1) {
1015  /* Even though we are only swapping out of i1 or i2,
1016  * we have to lock both of them along with the slot we
1017  * are swapping to, since at the end of this function,
1018  * i1 and i2 must be locked. */
1019  ob = (fb == i1) ? i2 : i1;
1020  lock_three(ti, fb, tb, ob);
1021  } else {
1022  lock_two(ti, fb, tb);
1023  }
1024 
1025  /* We plan to kick out fs, but let's check if it is still
1026  * there; there's a small chance we've gotten scooped by a
1027  * later cuckoo. If that happened, just... try again. Also
1028  * the slot we are filling in may have already been filled
1029  * in by another thread, or the slot we are moving from
1030  * may be empty, both of which invalidate the swap. */
1031  if (!eqfn(ti->buckets_[fb].keys[fs], from->key) ||
1032  ti->buckets_[tb].occupied[ts] ||
1033  !ti->buckets_[fb].occupied[fs]) {
1034  if (depth == 1) {
1035  unlock_three(ti, fb, tb, ob);
1036  } else {
1037  unlock_two(ti, fb, tb);
1038  }
1039  return false;
1040  }
1041 
1042  if (!is_simple) {
1043  ti->buckets_[tb].partials[ts] = ti->buckets_[fb].partials[fs];
1044  }
1045  ti->buckets_[tb].setKV(ts, ti->buckets_[fb].keys[fs], ti->buckets_[fb].vals[fs]);
1046  ti->buckets_[fb].eraseKV(fs);
1047  if (depth == 1) {
1048  // Don't unlock fb or ob, since they are needed in
1049  // cuckoo_insert. Only unlock tb if it doesn't unlock
1050  // the same bucket as fb or ob.
1051  if (lock_ind(tb) != lock_ind(fb) && lock_ind(tb) != lock_ind(ob)) {
1052  unlock(ti, tb);
1053  }
1054  } else {
1055  unlock_two(ti, fb, tb);
1056  }
1057  depth--;
1058  }
1059  return true;
1060  }

1062  /* run_cuckoo performs cuckoo hashing on the table in an attempt
1063  * to free up a slot on either i1 or i2. On success, the bucket
1064  * and slot that was freed up is stored in insert_bucket and
1065  * insert_slot. In order to perform the search and the swaps, it
1066  * has to unlock both i1 and i2, which can lead to certain
1067  * concurrency issues, the details of which are explained in the
1068  * function. If run_cuckoo returns ok (success), then the slot it
1069  * freed up is still locked. Otherwise it is unlocked. */
1070  cuckoo_status run_cuckoo(TableInfo *ti, const size_t i1, const size_t i2,
1071  size_t &insert_bucket, size_t &insert_slot) {
1072 
1073  CuckooRecord cuckoo_path[MAX_BFS_DEPTH+1];
1074 
1075  // We must unlock i1 and i2 here, so that cuckoopath_search
1076  // and cuckoopath_move can lock buckets as desired without
1077  // deadlock. cuckoopath_move has to look at either i1 or i2 as
1078  // its last slot, and it will lock both buckets and leave them
1079  // locked after finishing. This way, we know that if
1080  // cuckoopath_move succeeds, then the buckets needed for
1081  // insertion are still locked. If cuckoopath_move fails, the
1082  // buckets are unlocked and we try again. This unlocking does
1083  // present two problems. The first is that another insert on
1084  // the same key runs and, finding that the key isn't in the
1085  // table, inserts the key into the table. Then we insert the
1086  // key into the table, causing a duplication. To check for
1087  // this, we search i1 and i2 for the key we are trying to
1088  // insert before doing so (this is done in cuckoo_insert, and
1089  // requires that both i1 and i2 are locked). Another problem
1090  // is that an expansion runs and changes table_info, meaning
1091  // the cuckoopath_move and cuckoo_insert would have operated
1092  // on an old version of the table, so the insert would be
1093  // invalid. For this, we check that ti == table_info.load()
1094  // after cuckoopath_move, signaling to the outer insert to try
1095  // again if the comparison fails.
1096  unlock_two(ti, i1, i2);
1097 
1098  bool done = false;
1099  while (!done) {
1100  int depth = cuckoopath_search(ti, cuckoo_path, i1, i2);
1101  if (depth < 0) {
1102  break;
1103  }
1104 
1105  if (cuckoopath_move(ti, cuckoo_path, depth, i1, i2)) {
1106  insert_bucket = cuckoo_path[0].bucket;
1107  insert_slot = cuckoo_path[0].slot;
1108  assert(insert_bucket == i1 || insert_bucket == i2);
1109  assert(!ti->locks_[lock_ind(i1)].try_lock());
1110  assert(!ti->locks_[lock_ind(i2)].try_lock());
1111  assert(!ti->buckets_[insert_bucket].occupied[insert_slot]);
1112  done = true;
1113  break;
1114  }
1115  }
1116 
1117  if (!done) {
1118  return failure;
1119  } else if (ti != table_info.load()) {
1120  // Unlock i1 and i2 and signal to cuckoo_insert to try
1121  // again. Since we set the hazard pointer to be ti, this
1122  // check isn't susceptible to an ABA issue, since a new
1123  // pointer can't have the same address as ti.
1124  unlock_two(ti, i1, i2);
1125  return failure_under_expansion;
1126  }
1127  return ok;
1128  }

1247  /* cuckoo_insert tries to insert the given key-value pair into an
1248  * empty slot in i1 or i2, performing cuckoo hashing if necessary.
1249  * It expects the locks to be taken outside the function, but they
1250  * are released here, since different scenarios require different
1251  * handling of the locks. Before inserting, it checks that the key
1252  * isn't already in the table. cuckoo hashing presents multiple
1253  * concurrency issues, which are explained in the function. */
1254  cuckoo_status cuckoo_insert(const key_type &key, const mapped_type &val,
1255  const size_t hv, TableInfo *ti,
1256  const size_t i1, const size_t i2) {
1257  mapped_type oldval;
1258  int res1, res2;
1259  const partial_t partial = partial_key(hv);
1260  if (!try_add_to_bucket(ti, partial, key, val, i1, res1)) {
1261  unlock_two(ti, i1, i2);
1262  return failure_key_duplicated;
1263  }
1264  if (!try_add_to_bucket(ti, partial, key, val, i2, res2)) {
1265  unlock_two(ti, i1, i2);
1266  return failure_key_duplicated;
1267  }
1268  if (res1 != -1) {
1269  add_to_bucket(ti, partial, key, val, i1, res1);
1270  unlock_two(ti, i1, i2);
1271  return ok;
1272  }
1273  if (res2 != -1) {
1274  add_to_bucket(ti, partial, key, val, i2, res2);
1275  unlock_two(ti, i1, i2);
1276  return ok;
1277  }
1278 
1279  // we are unlucky, so let's perform cuckoo hashing
1280  size_t insert_bucket = 0;
1281  size_t insert_slot = 0;
1282  cuckoo_status st = run_cuckoo(ti, i1, i2, insert_bucket, insert_slot);
1283  if (st == failure_under_expansion) {
1284  /* The run_cuckoo operation operated on an old version of
1285  * the table, so we have to try again. We signal to the
1286  * calling insert method to try again by returning
1287  * failure_under_expansion. */
1288  return failure_under_expansion;
1289  } else if (st == ok) {
1290  assert(!ti->locks_[lock_ind(i1)].try_lock());
1291  assert(!ti->locks_[lock_ind(i2)].try_lock());
1292  assert(!ti->buckets_[insert_bucket].occupied[insert_slot]);
